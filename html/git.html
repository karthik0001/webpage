karthik@server1:~$ sudo apt install git
root@server1:/home/karthik# git config --global user.name "user.1"
root@server1:/home/karthik# git config --global user.email "user.1@karthik.ninja"

root@server1:/home/karthik# git config --list
user.name=user.1
user.email=user.1@karthik.ninja


root@server1:/home/karthik# mkdir samplerepo
root@server1:/home/karthik# cd samplerepo/

root@server1:/home/karthik/samplerepo# git init
Initialized empty Git repository in /home/karthik/samplerepo/.git/
root@server1:/home/karthik/samplerepo# ls -ltra
total 12
drwxr-xr-x 5 karthik karthik 4096 Jul 18 21:58 ..
drwxr-xr-x 3 root    root    4096 Jul 18 21:58 .
drwxr-xr-x 7 root    root    4096 Jul 18 21:58 .git


root@server1:/home/karthik/samplerepo# cat > index.html
Adding random file into the repo

root@server1:/home/karthik/samplerepo# ls
index.html
root@server1:/home/karthik/samplerepo# touch file1 file2
root@server1:/home/karthik/samplerepo#
root@server1:/home/karthik/samplerepo# echo "git add ."+"git add *"+"git add file1"
git add .+git add *+git add file1
root@server1:/home/karthik/samplerepo# git add *
root@server1:/home/karthik/samplerepo# git status
On branch master

No commits yet

Changes to be committed:
  (use "git rm --cached <file>..." to unstage)

        new file:   file1
        new file:   file2
        new file:   index.html

root@server1:/home/karthik/samplerepo# git commit -m "Initial Commit"
[master (root-commit) d812caf] Initial Commit
 3 files changed, 1 insertion(+)
 create mode 100644 file1
 create mode 100644 file2
 create mode 100644 index.html
root@server1:/home/karthik/samplerepo#
